/*pipe_select.c*/
#include <fcntl.h>
#include <stdio.h>
#include <unistd.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>
#include <errno.h>

#define FIFO1	"in1"
#define FIFO2	"in2"
#define MAX_BUFFER_SIZE	1024
#define IN_FILES	3
#define TIME_DELAY	60
#define MAX(a,b)	((a>b)?(a):(b))
int main(void)
{
	int fds[IN_FILES];
	char buf[MAX_BUFFER_SIZE];
	int i, res, real_read, maxfd;
	struct timeval tv;
	fd_set inset,tmp_inset;
	fds[0]=0;
	if (access(FIFO1, F_OK) == -1)
	{
		if((mkfifo(FIFO1, 0666)<0) && (errno != EEXIST))
		{
			printf("Cannot create fifo file\n");
			exit(1);
		}
	}
	if (access(FIFO2, F_OK) == -1)
	{
		if((mkfifo(FIFO2, 0666)<0) && (errno != EEXIST))
		{
			printf("Cannot create fifo file\n");
			exit(1);
		}
	}
	if((fds[1] = open (FIFO1, O_RDONLY|O_NONBLOCK))<0)
	{
		printf("Open in1 error\n");
		return 1;
	}
	if((fds[2] = open (FIFO2, O_RDONLY|O_NONBLOCK))<0)
	{
		printf("Open in2 error\n");
		return 1;
	}
	maxfd = MAX(MAX(fds[0],fds[1]),fds[2]);
	FD_ZERO(&inset); //清空套接字集合
	for (i=0; i<IN_FILES; i++)
	{
		FD_SET(fds[i],&inset);
	}
	FD_SET(0, &inset); //将标准输入（fd=0）加到文件表述符集合rfds中。
	tv.tv_sec = TIME_DELAY;
	tv.tv_usec = 0;
	while(FD_ISSET(fds[0],&inset) || FD_ISSET(fds[1],&inset) || FD_ISSET(fds[2],&inset))
	{ // FD_ISSET 用于测试指定的文件描述符是否在该集合中。
		tmp_inset = inset;
		//int select(int maxfdp,fd_set *readfds,fd_set *writefds,fd_set *errorfds,struct timeval *timeout); 
		//int maxfdp是一个整数值，是指集合中所有文件描述符的范围，即所有文件描述符的最大值加1
		//select 它能够监视我们需要监视的文件描述符的变化情况——读写或是异常
		res = select(maxfd + 1, &tmp_inset, NULL, NULL, &tv);
		switch(res)
		{
			case -1:
				{
					printf("Select error\n");
					return 1;
				}
			break;
			case 0:
				{
					printf("Time out\n");
					return 1;
				}
			break;
			default:
				{
					for (i =0; i < IN_FILES; i++) //判断需要监听的句柄是否发生了事件
					{
						if (FD_ISSET(fds[i], &tmp_inset))
						{
							memset(buf, 0, MAX_BUFFER_SIZE);
							//将数组buf中的所有数据清0
							real_read = read(fds[i], buf, MAX_BUFFER_SIZE);
							if(real_read < 0)
							{
								if (errno != EAGAIN) //这表明你在非阻塞模式下调用了阻塞操作
								{
									return 1;
								}
							}
							else if(!real_read)
							{
								close(fds[i]);
								FD_CLR(fds[i], &inset);	 //FD_CLR宏是用来从套接字集合中删除指定的套接字
							}
							else
							{
								if (i ==0)
								{
									if ((buf[0] == 'q') || (buf[0] == 'Q'))
									{
										return 1;
									}
								}
								else
								{
									buf[real_read] = '\0'; //增加的一个字节中存放字符"\0" 
									printf("%s", buf);
								}
							}
						}
					}
				}
				break;
		}
	}
}
